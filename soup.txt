document.querySelector("#add-zip-code-button").addEventListener('click', function() {
    var zip = document.querySelector("#zipcode").value,
        xhr,
        formdata,
        zipcode_regex = /(^\d{5}$)|(^\d{5}-\d{4}$)/;

    document.querySelector("#zipcode").classList.remove('invalid-zip');

    if(!zipcode_regex.test(zip)) {
        document.querySelector("#zipcode").classList.add('invalid-zip');
        return;
    }

    formdata = new FormData();
    formdata.append('operation', 'setZipCode');
    formdata.append('zip', zip);

    document.querySelector("#add-zip-code-button").disabled = true;

    xhr = new XMLHttpRequest();
    xhr.open('POST', _ADD_CART_AJAX_URL);
    xhr.onreadystatechange = function () {
        if(xhr.readyState == 4) {
            var response = JSON.parse(xhr.responseText);
            document.querySelector("#add-zip-code-button").disabled = false;

            if(xhr.status == 200) {
                if('updated' in response) {
                    document.querySelector("#item-add-zip-container").style.display = 'none';
                    document.querySelector("#zipcode").value = '';

                    document.querySelector("#item-zip-added-container").style.display = 'block';
                    document.querySelector("#current-zip").innerHTML = zip;
                    document.querySelector(".item-added-cart-button").href = _ADD_CART_MAIN_CART_PAGE;
                    document.querySelector(".item-added-cart-button").classList.remove('item-added-cart-button-disabled');
                }
            }
            else if(xhr.status == 400) {
                document.querySelector("#item-zip-error").innerHTML = response.message;
                document.querySelector("#item-zip-error").style.display = 'block';
                setTimeout(function() {
                    document.querySelector("#item-zip-error").style.display = 'none';
                }, 10000);
                document.querySelector(".item-added-cart-button").href = '#';
                document.querySelector(".item-added-cart-button").classList.add('item-added-cart-button-disabled');
            }
        }
    };
    xhr.send(formdata);
});

function addProduct(e) {
    var product_id = this.getAttribute("data-product-id"),
        is_outlet = (this.hasAttribute("data-outlet") ? this.getAttribute("data-outlet") : 0),
        quantity = (this.querySelector(".quantity") === null ? 1 : this.querySelector(".quantity").value),
        warranty = (this.querySelector(".warrantyChoiceRadio") === null ? "0" : this.querySelector(".warrantyChoiceRadio:checked").value),
        addonsCheckboxes = this.querySelectorAll("input[name='addons']"),
        xhr,
        formdata,
        addons = [],
        that = this;

    formdata = new FormData();
    formdata.append("operation", "addProduct");
    formdata.append("id", product_id);
    formdata.append("quantity", quantity);
    formdata.append("outlet", is_outlet);

    if (warranty != "0")
        formdata.append("warranty", warranty);

    addonsCheckboxes.forEach(function (element) {
        if (element.checked) {
            addons.push(element.value);
        }
    });

    if (addons.length)
        formdata.append("addons", addons.join("-"));

    this.querySelector(".product-model-add-cart").disabled = true;

    xhr = new XMLHttpRequest();
    xhr.open('POST', _ADD_CART_AJAX_URL);
    xhr.onreadystatechange = function () {
        if(xhr.readyState == 4) {
            var response = JSON.parse(xhr.responseText);
            that.querySelector(".product-model-add-cart").disabled = false;

            if(xhr.status == 200) {
                dataLayer.push({ ecommerce: null });  // Clear the previous ecommerce object.
					
						
                    dataLayer.push({
                        'event': 'addToCart',
                        'ecommerce': {
                            'add': {
                                "products": [
                                {
                                  
                                  "name": response.info.ModelDescription,
                                  "id": response.info.id,
                                  "price": response.info.Sale,
                                  "brand": response.info.CommonBrandName,
                                  "category": response.info.cSlug,
                                  "quantity": response.added
                                }
                              ]
                            }
                        }
                    });

                if('added' in response) {
                    document.querySelector("#item-added-popup-main .item-added-image img").setAttribute('src', response.info.thumb);
                    document.querySelector("#item-added-popup-main .item-added-brand").innerHTML = response.info.CommonBrandName;
                    document.querySelector("#item-added-popup-main .item-added-description").innerHTML = response.info.ModelDescription;
                    document.querySelector("#item-added-popup-main .item-added-model-value").innerHTML = response.info.SKU;
                    if (!Object.is(document.querySelector(".modal-epic-wrapper"), null)){
                        document.querySelector(".modal-epic-wrapper").setAttribute("pid", response.info.unique_item_id);    //This is for Pop-up Modal warranty selection; PDP; anupam
                    }

                    document.querySelector("#item-added-popup-main").style.display = 'block';
                    document.querySelector("#item-added-popup-main").setAttribute('data-item-id', response.item_id);

                    if(!document.querySelector(".item-added-cart-button").classList.contains('item-added-cart-button-disabled'))
                        document.querySelector(".item-added-cart-button").href = _ADD_CART_MAIN_CART_PAGE;

                    if(document.querySelector("#num-cart-items") !== null)
                        document.querySelector("#num-cart-items").innerHTML = response.num_cart_items;
                    if(document.querySelector("#cart-subtotal") !== null)
                        document.querySelector("#cart-subtotal").innerHTML = '$' + response.cart_subtotal;
                    if(document.querySelector("#cart-sidebar") !== null)
                        document.querySelector("#cart-sidebar").setAttribute('data-init', '0');
                }
            }
            else if(xhr.status == 400) {
                that.querySelector(".cart-add-error").innerHTML = response.message;
                that.querySelector(".cart-add-error").style.display = 'block';
                setTimeout(function() {
                    that.querySelector(".cart-add-error").style.display = 'none';
                }, 3000);
            }
        }
    };
    xhr.send(formdata);

    e.preventDefault();
    return false;
}

function ExpandableSection(toggle, expandable) {
    this.html = document.documentElement;
    this.toggle = toggle;
    this.expandable = expandable;

    this.toggle.addEventListener("click", this.handleClick.bind(this));
    this.html.style.scrollBehavior = "smooth";
}

ExpandableSection.prototype.handleClick = function (event) {
    this.expandable.classList.toggle("expanded");
    this.toggle.classList.toggle("expanded");

    if (!this.toggle.classList.contains("expanded")) {
        event.preventDefault();
    }
};
